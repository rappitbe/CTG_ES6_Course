Classes

Get Properties 

  const monsterHealth = Symbol();

  class Monster{
    constructor(name, health){
      this.name = name;
      this[monsterHealth] = health;
    }

    get isAlive(){
      return this[monsterHealth] > 0;
    }
  }


  let nessy = new Monster('Nessy', 100);
  console.log(nessy.isAlive); //Logs true

      }
